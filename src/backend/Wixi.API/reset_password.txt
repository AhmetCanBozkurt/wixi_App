using Microsoft.AspNetCore.Identity;
using Microsoft.Extensions.DependencyInjection;
using Wixi.Modules.Core.Domain.Entities;
using Wixi.Modules.Core.Infrastructure.Data;

var userEmail = "ahmetcanb785@gmail.com";
var newPassword = "Wixi123!";

var userManager = serviceProvider.GetRequiredService<UserManager<WixiUser>>();
var user = await userManager.FindByEmailAsync(userEmail);

if (user != null)
{
    var token = await userManager.GeneratePasswordResetTokenAsync(user);
    var result = await userManager.ResetPasswordAsync(user, token, newPassword);
    if (result.Succeeded)
    {
        Console.WriteLine($"Password for {userEmail} reset successfully to {newPassword}");
    }
    else
    {
        Console.WriteLine($"Failed to reset password: {string.Join(", ", result.Errors.Select(e => e.Description))}");
    }
}
else
{
    Console.WriteLine($"User {userEmail} not found.");
}
